function CreateTableFromJSON(mms,row){
var obj, dbParam, xmlhttp;
obj = { "limit":10 };
dbParam = JSON.stringify(obj);

param = mms;
xmlhttp = new XMLHttpRequest();
xmlhttp.onreadystatechange = function() {
if (this.readyState == 4 && this.status == 200) {
//document.getElementById("showData1").innerHTML = this.responseText;
var text = this.responseText;
var obj = JSON.parse(text);

// CREATE DYNAMIC TABLE.
var table = document.createElement("table");
table.setAttribute("class", "table holding");

// CREATE HTML TABLE HEADER ROW USING THE EXTRACTED HEADERS ABOVE.
var col = ["Status", "Library", "Location", "Call No."];
var tr = table.insertRow(-1); // TABLE ROW.
if(obj.status.length >0){
for (var i = 0; i < col.length; i++) {
var th = document.createElement("th"); // TABLE HEADER.
th.innerHTML = col[i];
tr.appendChild(th);
}
// ADD JSON DATA TO THE TABLE AS ROWS.
for (var i = 0; i < obj.status.length; i++) {
tr = table.insertRow(-1);
var tabCell = tr.insertCell(0);
if(obj.status[i]=="available"){
tabCell.setAttribute("class","green");
}
else{
tabCell.setAttribute("class","red");
}
var tabCell1 = tr.insertCell(1);
var tabCell2 = tr.insertCell(2);
var tabCell3 = tr.insertCell(3);
//var strStatus = obj.status[i];
//var strUpper = strStatus.toUpperCase();
//tabCell.innerHTML = strUpper;//moved to css
if(obj.status[i]!="available"){
//set display status
var strStatus = obj.status[i];
if(obj.status[i]=="unavailable"){
strStatus = 'not available';
}
if(obj.statusdesc[i]=="InProcess"){
tabCell.innerHTML = strStatus+'-'+'IN PROCESS';
}
else if(obj.statusdesc[i]=="ClaimedReturned"){
tabCell.innerHTML = strStatus+'-'+'CLAIMED RETURNED';
}
else if(obj.statusdesc[i]=="HoldShelf"){
tabCell.innerHTML = strStatus+'-'+'HOLD SHELF';
}
else if(obj.statusdesc[i]=="TechnicalMigration"){
tabCell.innerHTML = strStatus+'-'+'TECHNICAL MIGRATION';
}
//unavailable at all locations
else if(obj.statusdesc[i]=="unavailable"){
tabCell.innerHTML = strStatus;
}
//available at one of the locations
else if (obj.statusdesc[i]=="available_in_library"){
tabCell.innerHTML = strStatus;
}
else{
//tabCell.innerHTML = obj.status[i]+'-'+obj.statusdesc[i];
tabCell.innerHTML = strStatus+'-'+obj.statusdesc[i];
}
}
else{
tabCell.innerHTML = obj.status[i];
}
tabCell1.innerHTML = obj.location[i];
tabCell2.innerHTML = obj.stack[i];
if(obj.callno[i]==""){
//var result = str.link("http://lib.hkapa.edu/bib/".concat(mms));
//var url = "http://lib.hkapa.edu/bib/".concat(mms);
//tabCell3.innerHTML = '<a href="'+ url +'" target=_blank>CLICK for details </a>';
tabCell3.innerHTML = '-';
}
else{
tabCell3.innerHTML = obj.callno[i];
}
}
}
else{
//call function: get redirect link
GetRedirectLink(param,row);
//alert("No Status Retrieved. Please click view details.");
var th = document.createElement("th"); // TABLE HEADER.
th.innerHTML = "Status";
tr.appendChild(th);
tr = table.insertRow(-1);
var tabCell = tr.insertCell(0);

}

var divContainer = document.getElementById("showData"+row);
divContainer.innerHTML = "";
divContainer.appendChild(table);
}
};
